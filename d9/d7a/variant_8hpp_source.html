326  template<bool B, typename T0, typename T1, typename...Ts>
327  static void swap_impl( std::size_t current,
328  std::size_t index,
329  variant_union<B,T0,T1,Ts...>& lhs,
330  variant_union<B,T0,T1,Ts...>& rhs )
331  {
332  using std::swap;
333 
334  if( current == index ) {
335  swap( lhs.current, rhs.current );
336  } else {
337  swap_impl( current+1, index, lhs.next, rhs.next );
338  }
339  }

342  static void swap_impl( std::size_t,
343  std::size_t,
344  variant_union<B,T0>& lhs,
345  variant_union<B,T0>& rhs )
346  {
347  using std::swap;
348 
349  swap( lhs.current, rhs.current );
350  }

353  static void destroy_impl( std::size_t n, variant_union<false,T0>& storage )
354  {
355  storage.current.~T0();
356  }
357 
358  template<typename T0, typename T1, typename...Ts>
359  static void destroy_impl( std::size_t n, variant_union<false,T0,T1,Ts...>& storage )
360  {
361  if( n == 0 ) {
362  storage.current.~T0();
363  } else {
364  destroy_impl( n-1, storage.next );
365  }
