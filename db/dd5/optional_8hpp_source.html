80  template<typename T, typename U>
81  using optional_is_convertible = conjunction<
82  std::is_constructible<T, optional<U>&>,
83  std::is_constructible<T, const optional<U>&>,
84  std::is_constructible<T, optional<U>&&>,
85  std::is_constructible<T, const optional<U>&&>,
86  std::is_convertible<optional<U>&, T>,
87  std::is_convertible<const optional<U>&, T>,
88  std::is_convertible<optional<U>&&, T>,
89  std::is_convertible<const optional<U>&&, T>
90  >;
91 
92  template<typename T, typename U>
93  using optional_is_copy_convertible = conjunction<
94  optional_is_convertible<T,U>,
95  std::is_constructible<T, const U&>
96  >;
97 
98  template<typename T, typename U>
99  using optional_is_move_convertible = conjunction<
100  optional_is_convertible<T,U>,
101  std::is_constructible<T, U&&>
102  >;
103 
104  template<typename T, typename U>
105  using optional_is_value_convertible = conjunction<
106  std::is_constructible<T, U&&>,
107  negation<std::is_same<std::decay_t<U>,in_place_t>>,
108  negation<std::is_same<std::decay_t<U>,optional<T>>>
109  >;
110 
111  template<typename T, typename U>
112  using optional_is_convert_assignable = conjunction<
113  optional_is_convertible<T,U>,
114  std::is_assignable<T&, optional<U>&>,
115  std::is_assignable<T&, const optional<U>&>,
116  std::is_assignable<T&, optional<U>&&>,
117  std::is_assignable<T&, const optional<U>&&>
118  >;
119 
120  template<typename T, typename U>
121  using optional_is_copy_convert_assignable = conjunction<
122  optional_is_convert_assignable<T,U>,
123  std::is_constructible<T, const U&>,
124  std::is_assignable<T&, const U&>
125  >;
126 
127  template<typename T, typename U>
128  using optional_is_move_convert_assignable = conjunction<
129  optional_is_convert_assignable<T,U>,
130  std::is_constructible<T, U&&>,
131  std::is_assignable<T&, U&&>
132  >;
133 
134  template<typename T, typename U>
135  using optional_is_value_assignable = conjunction<
136  negation<std::is_same<std::decay_t<U>,optional<T>>>,
137  std::is_constructible<T,U>,
138  std::is_assignable<T,U>,
139  disjunction<
140  negation<std::is_same<std::decay_t<U>,T>>,
141  negation<std::is_scalar<T>>
142  >
