template<typename CharT, typename Traits = std::char_traits<CharT>, std::ptrdiff_t Extent = dynamic_extent>
class bit::core::basic_string_span< CharT, Traits, Extent >

A light-weight non-owning wrapper around strings.

Template Parameters
CharTthe type of character to wrap around
Traitsthe type of traits to use
ExtentThe maximum size of this string span (default: unbounded)
